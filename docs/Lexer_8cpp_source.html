192 bool Lexer::checkNumber()
193 {
194  if (!isdigit(currChar_)) {
195  return false;
196  }
197  // ASCII number - '0' = number
198  int val = currChar_ - '0';
199  int digitCounter = 1;
200  // Need to check zero first - special case
201  if (match('0')) {
202  if (isdigit(currChar_)) {
203  char errorMsg[MAX_ERROR_MSG_LEN] = {};
204  snprintf(errorMsg, MAX_ERROR_MSG_LEN, "ERROR: NUMBER CANNOT START WITH ZERO");
205  throw LangException(errorMsg, tokenStartPosition_);
206  } else {
207  currToken_ = Token(TokenType::IntegerLiteral, tokenStartPosition_, 0);
208  }
209  }
210  // Number multiple > 9
211  else {
212  nextChar();
213  while (isdigit(currChar_)) {
214  if (digitCounter > MAX_DIGITS) {
215  char errorMsg[MAX_ERROR_MSG_LEN] = {};
216  snprintf(errorMsg, MAX_ERROR_MSG_LEN, "ERROR: NUMBER TOO BIG");
217  throw LangException(errorMsg, tokenStartPosition_);
218  }
219  // Check for overflow using util function, need to cast beforehand - otherwise val becomes literal
220  checkOverflow(static_cast<long long>(val) * 10, currChar_ - '0');
221  val = val * 10 + (currChar_ - '0');
222  nextChar();
223  digitCounter++;
224  }
225  }
226  // float
227  if (match('.')) {
228  int exponent = 1;
229  if (!isdigit(currChar_)) {
230  currToken_ = Token(TokenType::FloatLiteral,tokenStartPosition_, static_cast<float>(val));
231  return true;
232  } else {
233  int fractionPart = currChar_ - '0';
234  nextChar();
235  digitCounter++;
236  while(isdigit(currChar_) && digitCounter < MAX_DIGITS) {
237  fractionPart = fractionPart * 10 + (currChar_ - '0');
238  nextChar();
239  digitCounter++;
240  exponent++;
241  }
242  if (digitCounter > MAX_DIGITS) {
243  char errorMsg[MAX_ERROR_MSG_LEN] = {};
244  snprintf(errorMsg, MAX_ERROR_MSG_LEN, "ERROR: NUMBER TOO BIG OCCURRED");
245  throw LangException(errorMsg, tokenStartPosition_);
246  }
247  float fraction = fractionPart / (pow(10, exponent));
248  currToken_ = Token(TokenType::FloatLiteral, tokenStartPosition_, val + fraction);
249  return true;
250  }
251  }
252  currToken_ = Token(TokenType::IntegerLiteral, tokenStartPosition_, val);
253  return true;
254 }
