8 void SemanticAnalyzerVisitor::visitProgram(Nodes::Program* program) {
9  if (program->getFunctions().empty()) {
10  throw LangException("Program must contain at least one function", program->getPosition(), __FUNCTION__ , __LINE__ );
11  }
12  if (program->getFunctions().find("main") == program->getFunctions().end()) {
13  throw LangException("Program must contain a main function", program->getPosition(), __FUNCTION__ , __LINE__ );
14  }
15 
16  symbolManager_.enterNewContext();
17  symbolManager_.enterNewScope();
18 
19  for (const auto & iter : program->getVariables()) {
20  iter.second->accept(*this);
21  }
22 
23  for (const auto& iter : program->getFunctions()) {
24  symbolManager_.insertSymbol(iter.first, SymbolTable::Symbol(true, iter.second->getType(), iter.second.get()));
25  }
26 
27  for (const auto & iter : program->getFunctions()) {
28  iter.second->accept(*this);
29  }
30 
31  symbolManager_.exitContext();
32 }

56 void SemanticAnalyzerVisitor::visitFunctionCall(Nodes::FunctionCall *fc) {
57  auto symbol = symbolManager_.getSymbol(fc->getIdentifier(), true);
58 
59  if (fc->getIdentifier() == "print") {
60  throw LangException("Cannot use print() as a r-value or argument!", fc->getPosition());
61  }
62  if (!symbol.has_value()) {
63  throw LangException("Function "+fc->getIdentifier()+" is not defined!", fc->getPosition());
64  }
65  if (!symbol.value().isFunction()) {
66  throw LangException("Cannot use variable as a function " + fc->getIdentifier(), fc->getPosition());
67  }
68  if (expectedType_ != std::nullopt) {
69  if (symbol.value().getType() != expectedType_) {
70  throw LangException("Invalid function type: "+fc->getIdentifier()+
71  "() returns: " + Nodes::identiferTypes[symbol.value().getType()] +
72  " but expected: " + Nodes::identiferTypes[expectedType_.value()], fc->getPosition());
73  }
74  }
75 
76  auto declaredArgs = symbol.value().getFunPointer().value()->getArgsDeclaration();
77  auto calledArgs = fc->getArgList();
78 
79  if (declaredArgs.has_value()) {
80  if (calledArgs.has_value()) {
81  if (declaredArgs.value().size() != calledArgs.value().size()) {
82  throw LangException("Invalid number of arguments in function call: " + fc->getIdentifier(), fc->getPosition());
83  }
84 
85  for (int i = 0; i < declaredArgs.value().size(); i++) {
86  auto previous = expectedType_;
87  expectedType_ = declaredArgs.value()[i]->getType();
88  calledArgs.value()[i]->accept(*this);
89  expectedType_ = previous;
90  }
91  } else {
92  throw LangException("The amount of args not a match with declaration: " + fc->getIdentifier(), fc->getPosition());
93  }
94  } else {
95  if (!calledArgs.has_value()) {
96  return;
97  }
98  throw LangException("The amount of args not a match with declaration: " + fc->getIdentifier(), fc->getPosition());
99  }
100 }
