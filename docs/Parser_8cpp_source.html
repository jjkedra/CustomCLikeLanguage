31  && !match(TokenType::DictToken)) {
32  return false;
33  } else {
34  idType type = INT;
35  if (currToken_.getType() == TokenType::FloatToken) {
36  type = FLOAT;

38  if (currToken_.getType() == TokenType::StringToken) {

42  type = DICT;
43  }
44

46 
47  if (!match(TokenType::Identifier)) {
48  throw LangException("No identifier after type", currToken_.getPosition());
49  }

52  auto lookUpVar = variables_.find(identifier);

55  }

57  if (lookUpFun != function_.end()) {
58  throw LangException("Function redefinition", currToken_.getPosition());

182 std::unique_ptr<Nodes::Dictionary> Parser::parseDictionary() {
183  idType type_key;
184  idType type_value;
185 
186  TokenType tt_key;
187  TokenType tt_value;
188 
189  std::vector<std::variant<std::unique_ptr<Nodes::Number>, std::unique_ptr<Nodes::String>>> value_key;
190  std::vector<std::variant<std::unique_ptr<Nodes::Number>, std::unique_ptr<Nodes::String>>> value_value;
191  size_t dict_len = 0;
192 
193  // [(int, int) ((1, 2), (4, 5)) ()];
194  if (currToken_.getType() == TokenType::OpenSquareBracket) {
195  getNextToken();
196  if (currToken_.getType() == TokenType::OpenParenthesis) {
197  getNextToken();
198  tt_key = dictionaryValueType(currToken_);
199  type_key = dictionaryKeyType(currToken_);
200  getNextToken();
201  if (currToken_.getType() != TokenType::Comma) {
202  throw LangException("Missing comma in dict type declaration", currToken_.getPosition());
203  }
204 
205  getNextToken();
206  tt_value = dictionaryValueType(currToken_);
207  type_value = dictionaryKeyType(currToken_);
208  getNextToken();
209  if (currToken_.getType() != TokenType::ClosingParenthesis) {
210  throw LangException("Missing closing parenthesis in dict type declaration", currToken_.getPosition());
211  }
212 
213  getNextToken();
214  if (currToken_.getType() != TokenType::OpenParenthesis) {
215  throw LangException("Missing open parenthesis in dict value declaration", currToken_.getPosition());
216  }
217  getNextToken();
218  while (true) {
219  // Case no init variables
220  if (currToken_.getType() != TokenType::OpenParenthesis) {
221  return std::make_unique<Nodes::Dictionary>(type_key, type_value, currToken_.getPosition(), std::move(value_key), std::move(value_value));
222  }
223  getNextToken();
224 
225  // Get key value
226  if (currToken_.getType() != tt_key) {
227  throw LangException("Invalid key type in dict value pair declaration", currToken_.getPosition());
228  }
229  if (tt_key == TokenType::IntegerLiteral || tt_key == TokenType::FloatLiteral) {
230  value_key.push_back(std::move(parseNumber()));
231  } else if (tt_key == TokenType::StringToken) {
232  value_key.push_back(std::move(parseString()));
233  } else {
234  throw LangException("Invalid literal key type in dict value pair declaration", currToken_.getPosition());
235  }
236 
237  // Check comma
238  if (currToken_.getType() != TokenType::Comma) {
239  throw LangException("Missing comma in dict value pair declaration", currToken_.getPosition());
240  }
241  getNextToken();
242 
243  // Get value value
244  if (currToken_.getType() != tt_value) {
245  throw LangException("Invalid value type in dict value pair declaration", currToken_.getPosition());
246  }
247  if (tt_value == TokenType::IntegerLiteral || tt_value == TokenType::FloatLiteral) {
248  value_value.push_back(std::move(parseNumber()));
249  } else if (tt_value == TokenType::StringToken) {
250  value_value.push_back(std::move(parseString()));
251  } else {
252  throw LangException("Invalid literal value type in dict value pair declaration", currToken_.getPosition());
253  }
254 
255  // Check closing parenthesis
256  if (currToken_.getType() != TokenType::ClosingParenthesis) {
257  throw LangException("Missing closing parenthesis in dict value pair declaration", currToken_.getPosition());
258  }
259  getNextToken();
260 
261  // Check for another pair or end of val
262  if (currToken_.getType() == TokenType::ClosingParenthesis) {
263  getNextToken();
264  break;
265  } else if (currToken_.getType() == TokenType::Comma) {
266  getNextToken();
267  } else {
268  throw LangException("Invalid symbol in dict value declaration", currToken_.getPosition());
269  }
270  }
271  // Check for dict function parenthesis
272  if (currToken_.getType() != TokenType::OpenParenthesis) {
273  throw LangException("Missing open parenthesis in dict function declaration", currToken_.getPosition());
274  }
275  getNextToken();
276 
277  // Check if there is function parameter
278  if (currToken_.getType() == TokenType::Identifier)
279  {
280  // Check if function exists
281  std::string function = std::get<std::string>(currToken_.getValue());
282  auto lookUpFun = function_.find(function);
283  if (lookUpFun == function_.end()) {
284  throw LangException("No such function to use as dictionary function", currToken_.getPosition());
285  }
286  getNextToken();
287  }
288 
289 
290  // Check for closing parenthesis
291  if (currToken_.getType() != TokenType::ClosingParenthesis) {
292  throw LangException("Missing closing parenthesis in dict function declaration", currToken_.getPosition());
293  }
294  getNextToken();
295 
296  // Check for closing square bracket parenthesis
297  if (currToken_.getType() != TokenType::ClosingSquareBracket) {
298  throw LangException("Missing closing bracket in dict declaration", currToken_.getPosition());
299  }
300  getNextToken();
301 
302  return std::make_unique<Nodes::Dictionary>(type_key, type_value, currToken_.getPosition(), std::move(value_key), std::move(value_value));
303  } else {
304  throw LangException("Missing open parenthesis in dict type declaration", currToken_.getPosition());
305  }
306  }
307  return nullptr;
308 }

506  if (!left) {
507  return nullptr;
508  }
509 
510  if (currToken_.getType() == TokenType::Equal ||
511  currToken_.getType() == TokenType::GreaterEqual ||
512  currToken_.getType() == TokenType::Greater ||
513  currToken_.getType() == TokenType::LessEqual ||
514  currToken_.getType() == TokenType::Less ||
515  currToken_.getType() == TokenType::NotEqual
516  ) {
517  relationalOperator op;
518  switch (currToken_.getType()) {
519  case TokenType::Equal:
520  op = relationalOperator::EQUAL;
521  break;
522  case TokenType::NotEqual:
523  op = relationalOperator::NOTEQUAL;
524  break;
525  case TokenType::Greater:
526  op = relationalOperator::GREATER;
527  break;
528  case TokenType::GreaterEqual:
529  op = relationalOperator::GREATEREQUAL;
530  break;
531  case TokenType::Less:
532  op = relationalOperator::LESS;
533  break;
534  case TokenType::LessEqual:
535  op = relationalOperator::LESSEQUAL;
536  break;
537  default:
538  return nullptr;
539  }
540  std::unique_ptr<Nodes::RelationalOperator> rop = std::make_unique<Nodes::RelationalOperator>(op, currToken_.getPosition());
541  getNextToken();
542  std::unique_ptr<Nodes::ArithmeticExpression> right = parseArithmetic();
543 
544  if (!right) {
545  throw LangException("Invalid relational expression", currToken_.getPosition());
546  }
547  return std::make_unique<Nodes::Expression>(std::move(left), std::move(rop), std::move(right), currToken_.getPosition());
548  }
